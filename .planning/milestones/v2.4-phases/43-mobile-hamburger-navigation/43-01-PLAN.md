```javascript
function updateNavForAuth(user) {
    const logoutBtn = document.getElementById('logoutBtn');
    const navLinks = document.querySelectorAll('.nav-link[data-route]');

    if (user) {
        if (logoutBtn) logoutBtn.style.display = 'block';
        const permissions = window.getCurrentPermissions?.();
        navLinks.forEach(link => {
            const route = link.getAttribute('data-route');
            const hasAccess = permissions?.tabs?.[route]?.access ?? true;
            link.style.display = hasAccess ? '' : 'none';
        });
        const dropdowns = document.querySelectorAll('.nav-dropdown[data-route]');
        dropdowns.forEach(dropdown => {
            const route = dropdown.getAttribute('data-route');
            const hasAccess = permissions?.tabs?.[route]?.access ?? true;
            dropdown.style.display = hasAccess ? '' : 'none';
        });
    } else {
        if (logoutBtn) logoutBtn.style.display = 'none';
        navLinks.forEach(link => { link.style.display = 'none'; });
        const dropdowns = document.querySelectorAll('.nav-dropdown[data-route]');
        dropdowns.forEach(dropdown => { dropdown.style.display = 'none'; });
    }
}
```

From app/router.js (function to update):

```javascript
function updateNavigation(path) {
    document.querySelectorAll('.nav-link').forEach(link => {
        link.classList.remove('active');
        const href = link.getAttribute('href');
        if (href && href.startsWith('#' + path)) {
            link.classList.add('active');
        }
    });
    const adminDropdown = document.querySelector('.nav-dropdown-trigger');
    if (adminDropdown) {
        if (path === '/admin') {
            adminDropdown.classList.add('active');
        } else {
            adminDropdown.classList.remove('active');
        }
    }
}
```

**index.html — replace the existing inline `<script>` block** (the one that contains `window.toggleAdminDropdown`, `window.setAdminSection`, the outside-click listener, and the hashchange listener) with an expanded version that adds hamburger menu control. The new script block must:

2. Add these new functions and listeners:

```javascript
// ---- Mobile hamburger menu ----

var _mobileMenuOpen = false;

window.toggleMobileMenu = function() {
    if (_mobileMenuOpen) {
        closeMobileMenu();
    } else {
        openMobileMenu();
    }
};

function openMobileMenu() {
    _mobileMenuOpen = true;
    var menu = document.getElementById('mobileNavMenu');
    var backdrop = document.getElementById('mobileNavBackdrop');
    var btn = document.getElementById('hamburgerBtn');
    if (menu) menu.classList.add('is-open');
    if (backdrop) backdrop.classList.add('is-open');
    if (btn) { btn.classList.add('is-open'); btn.setAttribute('aria-expanded', 'true'); }
    document.body.style.overflow = 'hidden'; // lock scroll
}

window.closeMobileMenu = function() {
    _mobileMenuOpen = false;
    var menu = document.getElementById('mobileNavMenu');
    var backdrop = document.getElementById('mobileNavBackdrop');
    var btn = document.getElementById('hamburgerBtn');
    if (menu) menu.classList.remove('is-open');
    if (backdrop) backdrop.classList.remove('is-open');
    if (btn) { btn.classList.remove('is-open'); btn.setAttribute('aria-expanded', 'false'); }
    document.body.style.overflow = ''; // restore scroll
};

// Close mobile menu when viewport widens to desktop (>=768px)
window.addEventListener('resize', function() {
    if (window.innerWidth >= 768 && _mobileMenuOpen) {
        closeMobileMenu();
    }
});

// Close mobile menu on navigation (hash changes)
window.addEventListener('hashchange', function() {
    if (_mobileMenuOpen) closeMobileMenu();
    // Keep existing admin dropdown close
    var menu = document.getElementById('adminDropdownMenu');
    if (menu) menu.classList.remove('open');
});

// Handle mobile nav item tap — navigate and close menu
window.mobileNavClick = function(event, el) {
    // Navigation happens via href; just close the menu after a tiny delay
    // so the browser can process the hash change
    closeMobileMenu();
};
```

NOTE: The existing `window.addEventListener('hashchange', ...)` that closes the admin dropdown must be merged/replaced with the new combined hashchange handler shown above (don't register two hashchange listeners — consolidate into one).
