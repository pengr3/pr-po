**saveField() (SERV-04):**
```javascript
async function saveField(fieldName, newValue) {
    if (window.canEditTab?.('services') === false) return;
    // Locked fields — never save
    if (['service_code', 'client_id', 'client_code', 'service_type'].includes(fieldName)) return;
    // Validate and sanitize
    const updateData = { [fieldName]: newValue };
    await updateDoc(doc(db, 'services', currentServiceDocId), updateData);
    // Fire-and-forget edit history
    recordEditHistory('services', currentServiceDocId, fieldName, currentService[fieldName], newValue).catch(console.error);
    currentService = { ...currentService, [fieldName]: newValue };
}
```

**saveDetailServicePersonnel() — called after adding/removing personnel:**
```javascript
async function saveDetailServicePersonnel() {
    const prevUserIds = normalizePersonnel(currentService).userIds;
    const newUserIds = selectedDetailPersonnel.map(p => p.id);
    const newNames = selectedDetailPersonnel.map(p => p.name);
    await updateDoc(doc(db, 'services', currentServiceDocId), {
        personnel_user_ids: newUserIds,
        personnel_names: newNames,
        personnel_user_id: null,
        personnel_name: null,
        personnel: null
    });
    await syncServicePersonnelToAssignments(currentService.service_code, prevUserIds, newUserIds);
    currentService.personnel_user_ids = newUserIds;
    currentService.personnel_names = newNames;
}
```
