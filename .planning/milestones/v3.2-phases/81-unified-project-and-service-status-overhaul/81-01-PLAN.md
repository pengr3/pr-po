Replace the dual `internal_status` + `project_status` UI in `projects.js` and `project-detail.js` with a single unified Status dropdown using the 10 new option values defined in 81-CONTEXT.md D-02.     Add a new helper function `rebuildStatusFilterOptions()` that scans `projectsData` for legacy `project_status` values not in `UNIFIED_STATUS_OPTIONS` and injects them as a separate `<optgroup label="Other (legacy)">` at the bottom of the `#projectStatusFilter` dropdown.     Place this function adjacent to `applyFilters()` (within the same module scope):

    ```javascript
    function rebuildStatusFilterOptions() {
        const select = document.getElementById('projectStatusFilter');
        if (!select) return;
        const previousValue = select.value;
        const legacyValues = new Set();
        for (const project of projectsData) {
            const v = project.project_status;
            if (v && !UNIFIED_STATUS_OPTIONS.includes(v)) {
                legacyValues.add(v);
            }
        }
        const unifiedHtml = UNIFIED_STATUS_OPTIONS
            .map(s => `<option value="${s}">${s}</option>`)
            .join('');
        const legacyHtml = legacyValues.size > 0
            ? `<optgroup label="Other (legacy)">${[...legacyValues].sort().map(v => `<option value="${escapeHTML(v)}">${escapeHTML(v)} (legacy)</option>`).join('')}</optgroup>`
            : '';
        select.innerHTML = `<option value="">All Statuses</option>${unifiedHtml}${legacyHtml}`;
        // Restore previous selection if it still exists (either unified or legacy)
        if (previousValue && [...select.options].some(o => o.value === previousValue)) {
            select.value = previousValue;
        }
    }
    ```
