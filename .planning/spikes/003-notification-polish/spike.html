  MRF_APPROVED: {
    label: 'MRF Approved',

    icon: `<svg viewBox="0 0 24 24" fill="none" stroke="currentColor"

</svg>`
  },
  MRF_REJECTED: {

    color: '#ef4444',
    icon: `<svg viewBox="0 0 24 24" ...>
  <path d="M6 18L18 6M6 6l12 12"/>
</svg>`
  },
  // ... 14 more entries
};

// In renderDropdownRows() — ZERO CHANGE NEEDED:
// The template already does: ${meta.icon}
// innerHTML parses the SVG string automatically.

// Pros: Self-contained. No index.html changes.
//       Each TYPE_META entry shows its own icon inline.
// Cons: Long strings in TYPE_META (~40 chars/path avg).
//       "Same icon" requires copy-paste if reused.
// 003b: Define symbols once in index.html, reference by ID

// ── index.html (added once, inside <body>) ──
<svg style="display:none">
  <defs>
    <symbol id="i-check" viewBox="0 0 24 24">
      <path stroke-linecap="round" stroke-linejoin="round"
            d="M4.5 12.75l6 6 9-13.5"/>
    </symbol>
    <symbol id="i-x" viewBox="0 0 24 24">...</symbol>
    <!-- 16 symbols total -->
  </defs>
</svg>

// ── TYPE_META in app/notifications.js ──
const TYPE_META = {
  MRF_APPROVED: {
    color: '#059669',
    icon: `<svg viewBox="0 0 24 24"><use href="#i-check"/></svg>`
  },
  PR_DECIDED: {
    color: '#059669',
    icon: `<svg viewBox="0 0 24 24"><use href="#i-check-circle"/></svg>`
  },
  // Shared symbols cost nothing — MRF_APPROVED and PR_DECIDED
  // can use the same check icon or different ones, zero duplication.
};

// ── renderDropdownRows() — ZERO CHANGE NEEDED ──

// Pros: Short, readable TYPE_META entries.
//       True single source for shared icons.
//       stroke/fill props on <symbol> apply to all uses.
// Cons: Requires modifying index.html.
//       <use href> cross-document usage not supported
//       (same-document is fine; external SVG files are not).

Key Observations

  • Visual consistency: All 16 SVG icons share identical stroke-width (1.75), linecap (round), linejoin (round) — the set feels unified. Text chars don't share any visual properties.
  • Emoji elimination: ⏳ (REGISTRATION_PENDING) and 📦 (PO_DELIVERED) were the worst offenders — emoji render at inconsistent sizes across Windows/Mac/mobile. Both now have proper stroked SVG icons.
  • Icon conflicts resolved: ✓ was shared by MRF_APPROVED / PR_DECIDED / TR_DECIDED. Now: simple check / check-circle / badge-check — visually distinct at 16px.
  • Dollar conflicts resolved: $ was shared by PROJECT_COST_CHANGED / RFP_PAID / COLLECTIBLE_CREATED. Now: dollar-circle / banknotes / coin — each tells a different story.
  • 003a vs 003b: Identical visual output. 003a is simpler — no index.html changes, TYPE_META is self-documenting. 003b is cleaner at scale (true symbol reuse). For this SPA's 16 types, 003a wins on simplicity.
