```javascript
async function saveField(fieldName, newValue) {
  // Permission guard
  if (window.canEditTab?.('projects') === false) return false;
  // Locked-fields rejection (project_code, client_id, client_code)
  if (['project_code', 'client_id', 'client_code'].includes(fieldName)) return false;
  // Validation (positive number for budget / contract_cost)
  if ((fieldName === 'budget' || fieldName === 'contract_cost') && newValue) { /* ... */ }

  // Normalize
  let valueToSave = newValue;
  if (fieldName === 'budget' || fieldName === 'contract_cost') {
    valueToSave = newValue ? parseFloat(newValue) : null;
  }

  // Skip-on-no-change
  const oldValue = currentProject[fieldName];
  const normalizedOld = (fieldName === 'budget' || fieldName === 'contract_cost')
    ? (oldValue != null ? parseFloat(oldValue) : null)
    : oldValue;
  if (normalizedOld === valueToSave) return true;

  try {
    await updateDoc(projectRef, { [fieldName]: valueToSave, updated_at: ... });
    recordEditHistory(...);

    // Existing NOTIF-11 (project_status change) — DO NOT modify
    if (fieldName === 'project_status' && NOTIF11_STATUS_WHITELIST.includes(valueToSave)) {
      createNotificationForUsers({ user_ids: recipients, type: PROJECT_STATUS_CHANGED, ... });
    }

    // <-- NOTIF-19 inserts here: parallel block for cost fields (budget, contract_cost)
    return true;
  } catch (error) { /* ... */ }
}
```
