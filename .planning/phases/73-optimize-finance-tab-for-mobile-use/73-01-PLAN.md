```css
    /* Card headers with action controls: stack vertically on mobile */
    .card-header {
        flex-direction: column;
        align-items: flex-start;
        gap: 0.5rem;
    }

    /* Right-side controls container goes full-width and wraps */
    .card-header > div {
        width: 100%;
        display: flex;
        flex-wrap: wrap;
        gap: 0.5rem;

        min-width: 0;
        min-height: 44px;
```
