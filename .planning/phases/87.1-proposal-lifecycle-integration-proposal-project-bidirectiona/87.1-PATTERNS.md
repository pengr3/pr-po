async function loadProposalCard(parentDocId, parentCollection) {
    try {
        const q = query(
            collection(db, 'proposals'),
            where('project_id', '==', parentDocId)
        );
        const snap = await getDocs(q);
        const el = document.getElementById('proposalInlineCard');
        if (!el) return; // view navigated away
        if (snap.empty) {
            el.innerHTML = `<div class="proposal-inline-card"><p style="color:#64748b;">No proposal linked yet.</p></div>`;
            return;
        }
        const proposal = { id: snap.docs[0].id, ...snap.docs[0].data() };
        el.innerHTML = renderInlineProposalCard(proposal);
    } catch (err) {
        console.error('[ProjectDetail] Proposal card load error:', err);
    }
}

**getDashboardMode() analog — filterProposalsForUser()** (new function, mirrors getDashboardMode role branching at lines 22–27):
```javascript
function filterProposalsForUser(allProposals) {
    const user = window.getCurrentUser?.();
    if (!user) return [];
    const role = user.role;
    if (role === 'super_admin') return allProposals;
    if (role === 'operations_admin' || role === 'operations_user') {
        const assignedCodes = window.getAssignedProjectCodes?.();
        return allProposals.filter(p => {
            if ((p.parent_collection || 'projects') !== 'projects') return false;
            if (assignedCodes === null) return true;
            return assignedCodes.includes(p.project_code);
        });
    }
    if (role === 'services_admin' || role === 'services_user') {
        const assignedCodes = window.getAssignedServiceCodes?.();
        return allProposals.filter(p => {
            if ((p.parent_collection || 'projects') !== 'services') return false;
            if (assignedCodes === null) return true;
            return assignedCodes.includes(p.project_code);
        });
    }
    return []; // finance, procurement_staff — no sub-tab
}
```
